Popover

Caso 1: carga normal de un contenido html.

<p><a href="#btn-legal-modal-content" id="btn-legal-modal" rel="popover" class="_btn">Mentions legales</a></p>

<div id="btn-legal-modal-content" class="hide"<
    <p><strong>Contenido del popover</strong></p>
</div<

jQuery(document).ready(function(){
    var oTabcontent = jQuery('#btn-legal-modal-content').html();
    jQuery('#btn-legal-modal-content').remove();

    jQuery('#btn-legal-modal').popover({
        'content': oTabcontent,
        'title': cTituloPopover + '<a class="close" onclick="jQuery(\'#btn-legal-modal\').popover(\'hide\');return false;" href="#close-modal">×</a>',
        'placement': 'top',
        'trigger': 'click'
    }).click(function(e){e.preventDefault();});
});
